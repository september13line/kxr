	}

		limit = { 
			has_government = radical_socialist
			has_country_leader = {
				name = "Амет Озенбашлы"
				ruling_only = yes
			}
		}

			OR = {
				has_democratic_socialist_government = yes 
				has_government = social_democrat
			}
		}
	}

		limit = { 
			has_government = social_democrat
		}
		has_completed_focus = CRM_Draft_Constitution
		RUS = { has_elected_government = yes }
	}
	CRM_lenin = no
	country_exists = RUS

}
