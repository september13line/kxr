IRE_has_michael_collins = {
	custom_trigger_tooltip = {
		tooltip = IRE_has_michael_collins
		OR = {
			has_country_leader = {
				name = "Майкл Коллинз"
				ruling_only = yes
			}
			has_country_leader = {
				name = "Майкл Д Коллинз"
				ruling_only = yes
			}
			AND = {
				has_country_flag = IRE_taoiseach_is_leader
				has_idea = IRE_Michael_Collins_hog_mli
			}
		}
	}
}
